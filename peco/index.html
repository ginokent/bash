#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#djeeno.github.io/sh' 
##
#djeeno.github.io/sh
##

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/sh/peco/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/common/")" || exit 1 ##httpGet##

echo "  PECO_VERSION=latest" | sh -c "${stderrPipeInfo:?}"
prog_name=peco
bin_dir=/usr/local/bin

if ! command -v "${prog_name:?}" 1>/dev/null; then
  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" = true ]; then
    bash -c "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/brew/")"
    commands_for_install="brew install ${prog_name:?}"
  elif command -v apt-get 1>/dev/null; then
    commands_for_install="apt-get update && apt-get install -y ${prog_name:?}"
  elif [ "${envIsLinux}" = true ] && [ "${envIsAMD64}" = true ]; then
    url=$(curl -LRsS https://github.com/peco/peco/releases | grep -o "href=\"[^\"]*linux_amd64[^\"]*" | sed "s|href=\"|https://github.com|" | head -n 1)
    commands_for_install="
      ${httpGet:?} \"${url:?}\" >/tmp/${prog_name:?}.tar.gz && (cd /tmp && tar xfz /tmp/${prog_name:?}.tar.gz 'peco_*/${prog_name:?}' && mv -v /tmp/peco_*/${prog_name:?} ${bin_dir:?}/${prog_name:?})
    "
  else
    echo "OS not supported" | sh -c "${stderrPipeError:?}"
    exit 1
  fi

  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" != true ] && command -v sudo 1>/dev/null && { [ ! -w "${bin_dir:?}" ] || { [ -e "${bin_dir:?}/${prog_name:?}" ] && [ ! -w "${bin_dir:?}/${prog_name:?}" ]; }; }; then
    sudo -E bash -cx "${commands_for_install:?}"
  else
    bash -cx "${commands_for_install:?}"
  fi

  if ! command -v "${prog_name:?}" 1>/dev/null; then
    echo "${prog_name:?} not installed" | sh -c "${stderrPipeError:?}"
    exit 1
  fi
fi

echo "${prog_name:?} installed" | sh -c "${stderrPipeOK:?}"
