#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#djeeno.github.io/sh' 
##
#djeeno.github.io/sh
##

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/sh/v0.1.2/asdf/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/v0.1.2/common/")" || exit 1 ##httpGet##

asdf_version=${ASDF_VERSION:=v0.7.8}
echo "  ASDF_VERSION=${ASDF_VERSION:-USE_DEFAULT}" | sh -c "${stderrPipeInfo:?}"

echo "install asdf to \"~/.asdf\"" | sh -c "${stderrPipeInfo:?}"
test -d ~/.asdf || git clone https://github.com/asdf-vm/asdf.git ~/.asdf --branch "${asdf_version}"

echo "append codes for asdf to \"~/.bashrc\"" | sh -c "${stderrPipeInfo:?}"
# shellcheck disable=SC2016,SC2088
grep -q '~/.asdf/' ~/.bashrc || printf '\n%s\n%s\n%s\n' '# asdf' 'test ! -f ~/.asdf/asdf.sh || . "$_"' 'test ! -f ~/.asdf/completions/asdf.bash || . "$_"' | tee -a ~/.bashrc | sh -c "${stderrPipeInfo:?}"
# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "update asdf" | sh -c "${stderrPipeInfo:?}"
asdf update

echo "update asdf plugins" | sh -c "${stderrPipeInfo:?}"
asdf plugin-update --all || true

echo "show asdf plugins current versions" | sh -c "${stderrPipeInfo:?}"
asdf current

echo "asdf installed" | sh -c "${stderrPipeOK:?}"
