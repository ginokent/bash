#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#djeeno.github.io/bash' 
##
#djeeno.github.io/bash
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/bash/releases/v0.1.0/asdf/python/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/bash/releases/v0.1.0/common/")" || exit 1 ##httpGet##

echo "  PYTHON_VERSION=${PYTHON_VERSION:-USE_DEFAULT}" | sh -c "${pipe_info:?}"

lang_name="python"
echo "install ${lang_name} via asdf" | sh -c "${pipe_info:?}"

command -v asdf 1>/dev/null || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/bash/releases/v0.1.0/asdf/")" || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | sh -c "${pipe_info:?}"
asdf plugin-add "${lang_name}" || true



# ERROR
#   zipimport.ZipImportError: can't decompress data; zlib not available
# High Sierra
#   xcode-select --install
# Mojave
#   sudo installer -pkg /Library/Developer/CommandLineTools/Packages/macOS_SDK_headers_for_macOS_10.14.pkg -target /
# shellcheck disable=SC2154
[ "${factIsDarwin}" = true ] && { brew install openssl@1.1 readline xz zlib; CONFIGURE_OPTS="--with-openssl=$(brew --prefix openssl@1.1)"; export CONFIGURE_OPTS; }
# shellcheck disable=SC2154
[ "${factIsUbuntu}" = true ] && sudo sh -c 'apt update; apt -y install build-essential libbz2-dev libffi-dev libsqlite3-dev libreadline-dev'



lang_version="${PYTHON_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^3\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}" | sh -c "${pipe_info:?}"
asdf install "${lang_name}" "${lang_version}"



python2_version="$(asdf list-all "${lang_name}" | grep -E '^2\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"

echo "install ${lang_name} ${python2_version}" | sh -c "${pipe_info:?}"
asdf install "${lang_name}" "${python2_version}"
# python2 系を global に利用する
echo "set ${lang_name} ${python2_version} to global" | sh -c "${pipe_info:?}"
asdf global "${lang_name}" "${python2_version}"



echo "run asdf ${lang_name} reshim" | sh -c "${pipe_info:?}"
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | sh -c "${pipe_info:?}"
asdf current "${lang_name}"
