#!/usr/bin/env bash set -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/sh/brew/ | bash

{ { { command -v curl&&HttpGet() { curl -LRsS "$1"; }; }||{ command -v wget&&HttpGet() { wget -qO- "$1"; }; }; }; }>/dev/null||{ printf "\033[31m%s\033[0m\n" "$(date +%Y-%m-%dT%H:%M:%S%z) [  error] curl or wget are required.";exit 1; }&&export HttpGet;[[ ${envIsLoaded:-false} = true ]]||{ eval "$(HttpGet https://djeeno.github.io/sh/common/)"; }||exit 1 ##HttpGet##

echo "  BREW_VERSION=latest" | stderrPipeInfo
prog_name=brew

if ! command -v "${prog_name:?}" 1>/dev/null; then
  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" = true ]; then
    command -v git 1>/dev/null || xcode-select --install
    bash -c "$(curl -LRsS https://raw.githubusercontent.com/Homebrew/install/master/install.sh)"
  else
    echo "OS not supported" | stderrPipeError
    exit 1
  fi

  if ! command -v "${prog_name:?}" 1>/dev/null; then
    echo "${prog_name:?} not installed" | stderrPipeError
    exit 1
  fi
fi

echo "${prog_name:?} installed" | stderrPipeOK
