#!/usr/bin/env bash set -E -e -o pipefail -u cat >/dev/null << '#djeeno.github.io/bash' 
##
#djeeno.github.io/bash
##

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )" && export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(sh -c "${httpGet} https://djeeno.github.io/bash/common/")" || exit 1  ##httpGetDoNotRemoveThisComment##

prog_name=asdf

Infoln "  ASDF_VERSION=${ASDF_VERSION:-DEFAULT_VERSION}"

Infoln "Install ${prog_name:?}"

if ! command -v git >/dev/null; then
  Run bash -c "${httpGet} https://djeeno.github.io/bash/git/ | bash"
fi

asdf_version="${ASDF_VERSION:=$(
  git ls-remote --quiet --refs --tags https://github.com/asdf-vm/asdf.git |
    grep -Eo "v[0-9]+\.[0-9]+\.[0-9]+" |
    sort -V |
    tail -n 1
)}"

if test ! -d ~/.asdf; then
  Run git clone https://github.com/asdf-vm/asdf.git ~/.asdf --branch "${asdf_version:?}"
fi

# shellcheck disable=SC2016,SC2088
if ! grep -q '~/.asdf/' ~/.bashrc; then
  Infoln "Install asdf setting to ~/.bashrc"
  {
    echo '# asdf'
    echo 'test ! -f ~/.asdf/asdf.sh || source "$_"'
    echo 'test ! -f ~/.asdf/completions/asdf.bash || source "$_"'
  } |
  Run tee -a ~/.bashrc | sh -c "${pipe_info:?}"
fi

# shellcheck disable=SC1090
test ! -f ~/.asdf/asdf.sh ||
  Run source ~/.asdf/asdf.sh
# shellcheck disable=SC1090
test ! -f ~/.asdf/completions/asdf.bash ||
  Run source ~/.asdf/completions/asdf.bash

Run asdf update

Run asdf plugin-update --all || true

Run asdf current || true

Noticeln "${prog_name:?} installed"
