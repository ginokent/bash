#!/usr/bin/env bash set -E -e -o pipefail -u cat >/dev/null << '#newtstat.github.io/bash' 
##
#newtstat.github.io/bash
##

# check interpreter
[[ "${interpreter:="$(ps h -p $$ -o args='' | cut -d' ' -f1)"}" =~ bash$ ]] || { printf '\033[31;1m%s\033[0m\n' "ERROR: ${interpreter:?}: interpreter not supported"; exit 1; }

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )" && export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(sh -c "${httpGet} https://newtstat.github.io/bash/common/")" || exit 1  ##httpGetDoNotRemoveThisComment##

prog_name=asdf

RecINFO "ASDF_VERSION=${ASDF_VERSION:-DEFAULT_VERSION}"

RecINFO "Install ${prog_name:?}"

if ! command -v git >/dev/null; then
  RecExec bash -c "${httpGet} https://newtstat.github.io/bash/git/ | bash"
fi

asdf_version="${ASDF_VERSION:=$(
  git ls-remote --quiet --refs --tags https://github.com/asdf-vm/asdf.git |
    grep -Eo "v[0-9]+\.[0-9]+\.[0-9]+" |
    sort -V |
    tail -n 1
)}"

if test ! -d ~/.asdf; then
  RecExec git clone https://github.com/asdf-vm/asdf.git ~/.asdf --branch "${asdf_version:?}"
fi

# shellcheck disable=SC2016,SC2088
if ! grep -q '~/.asdf/' ~/.bashrc >/dev/null 2>&1; then
  RecINFO "Install asdf setting to ~/.bashrc"
  RecExec touch ~/.bashrc
  {
    echo '# asdf'
    echo 'test ! -f ~/.asdf/asdf.sh || source "$_"'
    echo 'test ! -f ~/.asdf/completions/asdf.bash || source "$_"'
  } |
  RecExec tee -a ~/.bashrc | sh -c "${pipe_info:?}"
fi

# shellcheck disable=SC1090
test ! -f ~/.asdf/asdf.sh ||
  RecExec source ~/.asdf/asdf.sh
# shellcheck disable=SC1090
test ! -f ~/.asdf/completions/asdf.bash ||
  RecExec source ~/.asdf/completions/asdf.bash

RecExec asdf update

RecExec asdf plugin-update --all || true

RecExec asdf current || true

RecNOTICE "${prog_name:?} installed"
