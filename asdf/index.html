#!/usr/bin/env bash cat >/dev/null <<'# djeeno.github.io' 
##
# djeeno.github.io
##

export HTTP_CLIENT; HTTP_CLIENT="$( { command -v curl 1>/dev/null && printf "curl -LSs "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"
[ "${HTTP_CLIENT:?"curl or wget are required to run this script"}" ] || exit 1

# shellcheck disable=SC2154
[ "${envIsLoaded}" = true ] || eval "$(${HTTP_CLIENT} https://djeeno.github.io/common/)" || exit 1

asdf_version=${ASDF_VERSION:=v0.7.4}
echo "env:" | stderrDebug
echo "  ASDF_VERSION=${ASDF_VERSION:-USE_DEFAULT}" | stderrDebug

echo "install asdf to \"~/.asdf\"" | stderrDebug
test -d ~/.asdf || git clone https://github.com/asdf-vm/asdf.git ~/.asdf --branch "${asdf_version}"

echo "append codes for asdf to \"~/.bashrc\"" | stderrDebug
# shellcheck disable=SC2016,SC2088
grep -q '~/.asdf/' ~/.bashrc || printf '\n%s\n%s\n%s\n' '# asdf' 'test ! -f ~/.asdf/asdf.sh || . "$_"' 'test ! -f ~/.asdf/completions/asdf.bash || . "$_"' | tee -a ~/.bashrc | stderrDebug
# shellcheck disable=SC1090
. ~/.asdf/asdf.sh
# shellcheck disable=SC1090
. ~/.asdf/completions/asdf.bash

echo "update asdf" | stderrDebug
asdf update

echo "update asdf plugins" | stderrDebug
asdf plugin-update --all

echo "show asdf plugins current versions" | stderrDebug
asdf current

echo "正常にインストールされました。" | stderrOK
