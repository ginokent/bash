#!/usr/bin/env bash cat >/dev/null <<'# djeeno.github.io' 
##
# djeeno.github.io
##

set -e -o pipefail

# 以下コマンドを実行してインストール:
# Run the following command to install:
#   $ curl https://djeeno.gitlab.io/asdf/python/ | bash

lang_name="python"
: Install "${lang_name}" via asdf

# shellcheck disable=SC2153
lang_version="${PYTHON_VERSION}"
printf '\e[32m%s\e[0m\n' "# env:"
printf '\e[32m%s\e[0m\n' "#   PYTHON_VERSION: ${PYTHON_VERSION:-USE_DEFAULT}"

: Set HTTP Client
# shellcheck disable=SC2155
export HTTP_CLIENT="$( { command -v curl 1>/dev/null && printf "curl -LSs "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"
[ "${HTTP_CLIENT:?"curl or wget are required to run this script"}" ] || exit 1

: Run common
# shellcheck disable=SC2154
[ "${factIsLoaded}" = true ] || eval "$(${HTTP_CLIENT} https://djeeno.gitlab.io/common/)" || exit 1

: Install asdf
command -v asdf 1>/dev/null || eval "$(${HTTP_CLIENT} https://djeeno.gitlab.io/asdf/)" || exit 1

: Init asdf
. ~/.asdf/asdf.sh
. ~/.asdf/completions/asdf.bash

: FOR PYTHON
# ERROR
#   zipimport.ZipImportError: can't decompress data; zlib not available
# High Sierra
#   xcode-select --install
# Mojave
#   sudo installer -pkg /Library/Developer/CommandLineTools/Packages/macOS_SDK_headers_for_macOS_10.14.pkg -target /
# shellcheck disable=SC2154
[ "${factIsDarwin}" = true ] && { brew install openssl@1.1 readline xz zlib; CONFIGURE_OPTS="--with-openssl=$(brew --prefix openssl@1.1)"; export CONFIGURE_OPTS; }
# shellcheck disable=SC2154
[ "${factIsUbuntu}" = true ] && sudo sh -c 'apt update; apt -y install build-essential libbz2-dev libffi-dev libsqlite3-dev libreadline-dev'

: Install
printf '\e[32m%s\e[0m\n' "# Add ${lang_name} plugin"
asdf plugin-add "${lang_name}" || true

printf '\e[32m%s\e[0m\n' "# Install ${lang_name} ${lang_version:="$(asdf list-all "${lang_name}" | grep -E '^3\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"
asdf install "${lang_name}" "${lang_version}"
## python3 系を global に利用する
#printf '\e[32m%s\e[0m\n' "# Set ${lang_name} ${lang_version} to global"
#asdf global "${lang_name}" "${lang_version}"

printf '\e[32m%s\e[0m\n' "# Install ${lang_name} ${python2_version:="$(asdf list-all "${lang_name}" | grep -E '^2\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"
asdf install "${lang_name}" "${python2_version}"
# python2 系を global に利用する
printf '\e[32m%s\e[0m\n' "# Set ${lang_name} ${python2_version} to global"
asdf global "${lang_name}" "${python2_version}"

printf '\e[32m%s\e[0m\n' "# Run asdf ${lang_name} reshim"
asdf reshim "${lang_name}"

printf '\e[32m%s\e[0m\n' "# Show asdf ${lang_name} current version"
asdf current "${lang_name}"
