#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/sh/asdf/kubetail/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/common/")" || exit 1 ##httpGet##

echo "  PROTOC_VERSION=${KUBETAIL_VERSION:-USE_DEFAULT}" | sh -c "${stderrPipeInfo:?}"

lang_name="protoc"
echo "install ${lang_name} via asdf" | sh -c "${stderrPipeInfo:?}"

asdf --version 1>/dev/null 2>&1 || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/asdf/")" || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | sh -c "${stderrPipeInfo:?}"
asdf plugin-add "${lang_name}" https://github.com/paxosglobal/asdf-protoc.git || true

lang_version="${KUBETAIL_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}" | sh -c "${stderrPipeInfo:?}"
asdf install "${lang_name}" "${lang_version}"

echo "set ${lang_name} ${lang_version} to global" | sh -c "${stderrPipeInfo:?}"
asdf global "${lang_name}" "${lang_version}"

echo "run asdf ${lang_name} reshim" | sh -c "${stderrPipeInfo:?}"
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | sh -c "${stderrPipeInfo:?}"
asdf current "${lang_name}"
