#!/usr/bin/env bash set -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/sh/asdf/golang/ | bash

export httpGet; httpGet="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/common/")" || exit 1 ##httpGet##

echo "  GOLANG_VERSION=${GOLANG_VERSION:-USE_DEFAULT}" | stderrPipeInfo

lang_name="golang"
echo "install ${lang_name} via asdf" | stderrPipeInfo

command -v asdf 1>/dev/null || source <(HttpGet https://djeeno.github.io/sh/asdf/) || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | stderrPipeInfo
asdf plugin-add "${lang_name}" || true

lang_version="${GOLANG_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}" | stderrPipeInfo
asdf install "${lang_name}" "${lang_version}"

echo "set ${lang_name} ${lang_version} to global" | stderrPipeInfo
asdf global "${lang_name}" "${lang_version}"

echo "run asdf ${lang_name} reshim" | stderrPipeInfo
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | stderrPipeInfo
asdf current "${lang_name}"



echo "add GOPATH settings to ~/.bashrc" | stderrPipeInfo
if ! grep -q GOPATH ~/.bashrc; then
cat >>~/.bashrc << 'BASHRC'
if command -v go 1>/dev/null; then
  export GOROOT="$(go env GOROOT)"
  export GOPATH=$HOME/go
  export PATH=$PATH:$GOROOT/bin:$GOPATH/bin
elif [ -d /usr/local/go ]; then
  export GOROOT="$(/usr/local/go/bin/go env GOROOT)"
  export GOPATH=$HOME/go
  export PATH=$PATH:$GOROOT/bin:$GOPATH/bin
fi
BASHRC
fi
