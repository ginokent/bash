#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/sh/asdf/redis/ | bash

export httpGet; httpGet="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/common/")" || exit 1 ##httpGet##

echo "  REDIS_VERSION=${REDIS_VERSION:-USE_DEFAULT}" | sh -c "exec ${stderrPipeInfo:?}"

lang_name="redis"
echo "install ${lang_name} via asdf" | sh -c "exec ${stderrPipeInfo:?}"

command -v asdf 1>/dev/null || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/asdf/")" || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | sh -c "exec ${stderrPipeInfo:?}"
asdf plugin-add "${lang_name}" || true

lang_version="${REDIS_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}" | sh -c "exec ${stderrPipeInfo:?}"
asdf install "${lang_name}" "${lang_version}"

echo "set ${lang_name} ${lang_version} to global" | sh -c "exec ${stderrPipeInfo:?}"
asdf global "${lang_name}" "${lang_version}"

echo "run asdf ${lang_name} reshim" | sh -c "exec ${stderrPipeInfo:?}"
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | sh -c "exec ${stderrPipeInfo:?}"
asdf current "${lang_name}"
