#!/usr/bin/env bash cat >/dev/null <<'#djeeno.github.io' 
##
#djeeno.github.io
##

set -e -o pipefail

# 以下コマンドを実行してインストール:
# Run the following command to install:
#   $ curl https://djeeno.gitlab.io/asdf/nodejs/ | bash

lang_name="nodejs"
: Install "${lang_name}" via asdf

# shellcheck disable=SC2153
lang_version="${NODEJS_VERSION}"
printf '\e[32m%s\e[0m\n' "# env:"
printf '\e[32m%s\e[0m\n' "#   NODEJS_VERSION: ${NODEJS_VERSION:-USE_DEFAULT}"

: Set HTTP Client
# shellcheck disable=SC2155
export http_get="$( { command -v curl 1>/dev/null && printf "curl -LSs "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"
[ "${http_get:?"curl or wget are required to run this script"}" ] || exit 1

: Run common
# shellcheck disable=SC2154
[ "${factIsLoaded}" = true ] || eval "$(${http_get} https://djeeno.gitlab.io/common/)" || exit 1

: Install asdf
command -v asdf 1>/dev/null || eval "$(${http_get} https://djeeno.gitlab.io/asdf/)" || exit 1

: Init asdf
. ~/.asdf/asdf.sh
. ~/.asdf/completions/asdf.bash

: FOR NODEJS
printf '\e[32m%s\e[0m\n' "# Install dependencies for nodejs"
(! uname -s | grep -q Darwin) || brew install gnupg
# If GnuPG keys are not imported, import from bash script
if [ "$(grep -Eo '[0-9A-Z]{40}' ~/.asdf/plugins/nodejs/bin/import-release-team-keyring | while read -r LINE; do gpg --list-keys | grep -q "${LINE}"; echo $?; done | sort -u)" != "0" ]; then
  bash ~/.asdf/plugins/nodejs/bin/import-release-team-keyring
fi

: Install
printf '\e[32m%s\e[0m\n' "# Add ${lang_name} plugin"
asdf plugin-add "${lang_name}" || true

printf '\e[32m%s\e[0m\n' "# Install ${lang_name} ${lang_version:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"
asdf install "${lang_name}" "${lang_version}"

printf '\e[32m%s\e[0m\n' "# Set ${lang_name} ${lang_version} to global"
asdf global "${lang_name}" "${lang_version}"

printf '\e[32m%s\e[0m\n' "# Run asdf ${lang_name} reshim"
asdf reshim "${lang_name}"

printf '\e[32m%s\e[0m\n' "# Show asdf ${lang_name} current version"
asdf current "${lang_name}"
