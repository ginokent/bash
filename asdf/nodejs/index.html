#!/usr/bin/env bash set -E -o pipefail cat >/dev/null <<'#djeeno.github.io' 
##
#djeeno.github.io
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/asdf/golang/ | bash

export http_get; http_get="$( { command -v curl 1>/dev/null && printf "curl -LSs "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; [ "${http_get:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(${http_get} https://djeeno.github.io/common/)" || exit 1

echo "env:" | stderrPipeInfo
echo "  NODEJS_VERSION: ${NODEJS_VERSION:-USE_DEFAULT}" | stderrPipeInfo

lang_name="nodejs"
echo "install ${lang_name} via asdf" | stderrPipeInfo

command -v asdf 1>/dev/null || eval "$(${http_get} https://djeeno.gitlab.io/asdf/)" || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | stderrPipeInfo
asdf plugin-add "${lang_name}" || true

echo "install dependencies for nodejs" | stderrPipeInfo
(! uname -s | grep -q Darwin) || brew install gnupg
# If GnuPG keys are not imported, import from bash script
if [ "$(grep -Eo '[0-9A-Z]{40}' ~/.asdf/plugins/nodejs/bin/import-release-team-keyring | while read -r LINE; do gpg --list-keys | grep -q "${LINE}"; echo $?; done | sort -u)" != "0" ]; then
  bash ~/.asdf/plugins/nodejs/bin/import-release-team-keyring
fi

lang_version="${NODEJS_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}"
asdf install "${lang_name}" "${lang_version}"

echo "set ${lang_name} ${lang_version} to global" | stderrPipeInfo
asdf global "${lang_name}" "${lang_version}"

echo "run asdf ${lang_name} reshim" | stderrPipeInfo
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | stderrPipeInfo
asdf current "${lang_name}"
