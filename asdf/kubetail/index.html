#!/usr/bin/env bash set -E -e -o pipefail cat >/dev/null << '#djeeno.github.io/bash' 
##
#djeeno.github.io/bash
##

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(bash -c "exec ${httpGet} https://djeeno.github.io/bash/common/")" || exit 1  ##httpGetDoNotRemoveThisComment##

LANG_NAME=KUBETAIL
lang_name=kubetail

LANG_VERSION_KEY="${LANG_NAME}_VERSION"
Debugln "  ${LANG_VERSION_KEY}=${!LANG_VERSION_KEY:-DEFAULT_VERSION}"

Infoln "Start to install ${lang_name:?}"

command -v asdf 1>/dev/null ||
  eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/bash/asdf/")" ||
  exit 1

Run asdf plugin-add "${lang_name}" https://github.com/janpieper/asdf-kubetail.git || true

lang_version="${!LANG_VERSION_KEY:="$(
  asdf list-all "${lang_name}" |
    grep -Eo '^[0-9]+\.[0-9]+\.[0-9]+$' |
    sort -V |
    tail -n 1
)"}"

Run asdf install "${lang_name}" "${lang_version}"

Run asdf global "${lang_name}" "${lang_version}"

Run asdf reshim "${lang_name}"

Run asdf current "${lang_name}"
