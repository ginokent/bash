#!/usr/bin/env bash set -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to execute this script:
#   $ curl https://sh.djeeno.com/bbr/ | bash

export http_get; http_get="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"
[ "${http_get:?"curl or wget are required to run this script"}" ] || exit 1
# shellcheck disable=SC2154
[ "${envIsLoaded}" = true ] || eval "$(${http_get} https://sh.djeeno.com/common/)" || exit 1

echo "  BBR_VERSION=USE_DEFAULT" | stderrPipeInfo

if ! command -v "${prog_name:?}" 1>/dev/null; then
  # shellcheck disable=SC2154
  if [ "${envIsLinux}" = true ]; then
    ##
    # 更新用コマンド
    ##
    # curl -LRsS https://raw.githubusercontent.com/teddysun/across/master/bbr.sh -o ~/sh.djeeno.com/bbr/bbr.sh
    ##
    commands_for_install="
      ${http_get:?} https://sh.djeeno.com/bbr/bbr.sh
    "
  else
    echo "OS not supported" | stderrPipeError
    exit 1
  fi

  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" != true ] && command -v sudo 1>/dev/null; then
    sudo -E bash -cx "${commands_for_install:?}"
  else
    bash -cx "${commands_for_install:?}"
  fi
fi
