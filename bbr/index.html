#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#djeeno.github.io/bash' 
##
#djeeno.github.io/bash
##

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/bash/bbr/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(bash -c "exec ${httpGet} https://djeeno.github.io/bash/common/")" || exit 1  ##httpGet##

echo "  BBR_VERSION=USE_DEFAULT" | sh -c "${pipe_info:?}"

if ! command -v "${prog_name:?}" 1>/dev/null; then
  # shellcheck disable=SC2154
  if [ "${EnvIsLinux}" = true ]; then
    ##
    # 更新用コマンド
    ##
    # curl -LRsS https://raw.githubusercontent.com/teddysun/across/master/bbr.sh -o ~/djeeno.github.io/bash/bbr/bbr.sh
    ##
    commands_for_install="
      ${httpGet:?} https://djeeno.github.io/bash/bbr/bbr.sh
    "
  else
    echo "OS not supported" | sh -c "${pipe_error:?}"
    exit 1
  fi

  # shellcheck disable=SC2154
  if [ "${EnvIsDarwin}" != true ] && command -v sudo 1>/dev/null; then
    sudo -E bash -cx "${commands_for_install:?}"
  else
    bash -cx "${commands_for_install:?}"
  fi
fi
