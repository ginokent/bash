#!/usr/bin/env bash set -Eeu -o pipefail cat >/dev/null << '#newtstat.github.io/bash' 
##
#newtstat.github.io/bash
##

# Run the following command to execute this script:
#   $ curl https://newtstat.github.io/bash/direnv/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )" && export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(sh -c "${httpGet} https://newtstat.github.io/bash/common/")" || exit 1  ##httpGetDoNotRemoveThisComment##

[[ ${TMUX_VERSION:-} ]] || export TMUX_VERSION=3.2
URL="https://github.com/tmux/tmux/archive/refs/tags/${TMUX_VERSION:?}.tar.gz"
FILE="tmux-${TMUX_VERSION:?}.tar.gz"

post_hook_command="if ! grep -q git-completion.bash ~/.tmux.conf; then
tee -a ~/.tmux.conf <<'EOF'
set -g mouse on
bind-key -T copy-mode    WheelUpPane   send -N1 -X scroll-up
bind-key -T copy-mode-vi WheelUpPane   send -N1 -X scroll-up
bind-key -T copy-mode    WheelDownPane send -N1 -X scroll-down
bind-key -T copy-mode-vi WheelDownPane send -N1 -X scroll-down
EOF
fi"

# shellcheck disable=SC2154
if [ "${EnvIsDarwin-}" = true ]; then
  command -v brew 1>/dev/null || bash -c "$(${httpGet} https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
  utilInstall \
    "! command -v tmux" \
    "brew install tmux" \
    "command -v brew 1>/dev/null || bash -c \"\$(${httpGet} https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)\"" \
    "${post_hook_command:?}"
elif [ "${EnvIsCentOS-}" = true ] || [ "${EnvIsRockyLinux-}" = true ] || [ "${EnvIsAmazonLinux-}" = true ]; then
  utilInstall \
    "! ls /usr/local/bin/tmux" \
    "mkdir -p /usr/local/src && cd /usr/local/src && wget -cSv ${URL:?} -O ${FILE:?} && tar xzf ${FILE:?} && cd $(basename -s .tar.gz "${FILE:?}") && sh autogen.sh && ./configure && make && make install && cd .. && tmux -V" \
    "yum install -y automake bison gcc gzip less libevent-devel ncurses-devel make tar wget" \
    "${post_hook_command:?}"
elif [ "${EnvIsUbuntu-}" = true ]; then
  utilInstall \
    "! ls /usr/local/bin/tmux" \
    "mkdir -p /usr/local/src && cd /usr/local/src && wget -cSv ${URL:?} -O ${FILE:?} && tar xzf ${FILE:?} && cd $(basename -s .tar.gz "${FILE:?}") && sh autogen.sh && ./configure && make && make install && cd .. && tmux -V" \
    "apt-get update -qy && DEBIAN_FRONTEND=noninteractive apt-get install -qy automake bison gcc gzip less libevent-dev libncurses-dev make pkg-config tar wget" \
    "${post_hook_command:?}"
else
  RecCritical "OS not supported"
  exit 1
fi
