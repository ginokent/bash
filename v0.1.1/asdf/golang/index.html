#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#djeeno.github.io/sh' 
##
#djeeno.github.io/sh
##

# Run the following command to install:
#   $ curl https://djeeno.github.io/sh/v0.1.1/asdf/golang/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/v0.1.1/common/")" || exit 1 ##httpGet##

echo "  GOLANG_VERSION=${GOLANG_VERSION:-USE_DEFAULT}" | sh -c "${pipe_info:?}"

lang_name="golang"
echo "install ${lang_name} via asdf" | sh -c "${pipe_info:?}"

asdf --version 1>/dev/null 2>&1 || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/v0.1.1/asdf/")" || exit 1

# shellcheck disable=SC1090
. ~/.asdf/asdf.sh && . ~/.asdf/completions/asdf.bash

echo "add ${lang_name} plugin" | sh -c "${pipe_info:?}"
asdf plugin-add "${lang_name}" || true

lang_version="${GOLANG_VERSION:="$(asdf list-all "${lang_name}" | grep -E '^[0-9]+\.[0-9]+\.[0-9]+$' | sort -V | tail -n 1)"}"

echo "install ${lang_name} ${lang_version}" | sh -c "${pipe_info:?}"
asdf install "${lang_name}" "${lang_version}"

echo "set ${lang_name} ${lang_version} to global" | sh -c "${pipe_info:?}"
asdf global "${lang_name}" "${lang_version}"

echo "run asdf ${lang_name} reshim" | sh -c "${pipe_info:?}"
asdf reshim "${lang_name}"

echo "show asdf ${lang_name} current version" | sh -c "${pipe_info:?}"
asdf current "${lang_name}"



echo "add GOPATH settings to ~/.bashrc" | sh -c "${pipe_info:?}"
if ! grep -q GOPATH ~/.bashrc; then
cat >>~/.bashrc << 'BASHRC'
if command -v go 1>/dev/null; then
  export GOROOT="$(go env GOROOT)"
  export GOPATH=$HOME/go
  export PATH=$PATH:$GOROOT/bin:$GOPATH/bin
elif [ -d /usr/local/go ]; then
  export GOROOT="$(/usr/local/go/bin/go env GOROOT)"
  export GOPATH=$HOME/go
  export PATH=$PATH:$GOROOT/bin:$GOPATH/bin
fi
BASHRC
fi
