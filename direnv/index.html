#!/usr/bin/env bash set -E -o pipefail cat >/dev/null <<'#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to execute this script:
#   $ curl https://sh.djeeno.com/direnv/ | bash

export http_get; http_get="$( { command -v curl 1>/dev/null && printf "curl -LSs "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"
[ "${http_get:?"curl or wget are required to run this script"}" ] || exit 1
# shellcheck disable=SC2154
[ "${envIsLoaded}" = true ] || eval "$(${http_get} https://sh.djeeno.com/common/)" || exit 1

echo "env:" | stderrPipeInfo
echo "  DIRENV_VERSION: ${DIRENV_VERSION:-USE_DEFAULT}" | stderrPipeInfo
direnv_version=${DIRENV_VERSION:=2.20.0}

if ! command -v direnv 1>/dev/null; then
  commands_for_install="
    curl -LR \"https://github.com/direnv/direnv/releases/download/v${direnv_version}/direnv.\$(uname -s | tr A-Z a-z)-amd64\" -o /usr/local/bin/direnv
    chmod +x /usr/local/bin/direnv
  "

  if [ ! -w /usr/local/bin ] || [ ! -w /usr/local/bin/direnv ]; then
    sudo -E bash -c "eval ${commands_for_install}"
  else
    eval "${commands_for_install}"
  fi

  if ! command -v direnv 1>/dev/null; then
    echo "direnv not installed" | stderrPipeError
    exit 1
  fi
fi

echo "direnv installed" | stderrPipeOK
