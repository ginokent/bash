#!/usr/bin/env bash set -e -E -o pipefail cat >/dev/null << '#sh.djeeno.com' 
##
#sh.djeeno.com
##

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/sh/direnv/ | bash

export httpGet; httpGet="$( { command -v curl 1>/dev/null && printf "curl -LRsS "; } || { command -v wget 1>/dev/null && printf "wget -qO- "; } )"; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${envIsLoaded:-false}" = true ] || eval "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/common/")" || exit 1 ##httpGet##

echo "  DIRENV_VERSION=latest" | sh -c "${stderrPipeInfo:?}"
prog_name=direnv
bin_dir=/usr/local/bin

if ! command -v "${prog_name:?}" 1>/dev/null; then
  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" = true ]; then
    bash -c "$(sh -c "exec ${httpGet} https://djeeno.github.io/sh/brew/")"
    commands_for_install="brew install ${prog_name:?}"
  elif [ "${envIsLinux}" = true ] && [ "${envIsAMD64}" = true ]; then
    url=$(curl -LRsS https://github.com/direnv/direnv/releases/ | grep -o "href=\"/direnv/direnv/releases/download/v[0-9]*\.[0-9]*\.[0-9]*/direnv.$(uname -s | tr '[:upper:]' '[:lower:]')-amd64" | sed "s|href=\"|https://github.com|" | head -n 1)
    commands_for_install="
      ${httpGet:?} \"${url:?}\" >${bin_dir:?}/${prog_name:?} && chmod +x ${bin_dir:?}/${prog_name:?}
    "
  else
    echo "OS not supported" | sh -c "${stderrPipeError:?}"
    exit 1
  fi

  # shellcheck disable=SC2154
  if [ "${envIsDarwin}" != true ] && command -v sudo 1>/dev/null && { [ ! -w "${bin_dir:?}" ] || { [ -e "${bin_dir:?}/${prog_name:?}" ] && [ ! -w "${bin_dir:?}/${prog_name:?}" ]; }; }; then
    sudo -E bash -cx "${commands_for_install:?}"
  else
    bash -cx "${commands_for_install:?}"
  fi

  if ! command -v "${prog_name:?}" 1>/dev/null; then
    echo "${prog_name:?} not installed" | sh -c "${stderrPipeError:?}"
    exit 1
  fi
fi

echo "${prog_name:?} installed" | sh -c "${stderrPipeOK:?}"
