#!/usr/bin/env bash set -E -e -o pipefail cat >/dev/null << '#djeeno.github.io/bash' 
##
#djeeno.github.io/bash

# Run the following command to execute this script:
#   $ curl https://djeeno.github.io/bash/peco/ | bash

httpGet="$( { command -v curl 1>/dev/null && printf "curl -fLRSs"; } || { command -v wget 1>/dev/null && printf "wget -O- -q"; } )"; export httpGet; [ "${httpGet:?"curl or wget are required"}" ] || exit 1; [ "${EnvIsLoaded:-false}" = true ] || eval "$(bash -c "exec ${httpGet} https://djeeno.github.io/bash/common/")" || exit 1  ##httpGetDoNotRemoveThisComment##

__main () { (
  if [ "${EnvIsDarwin:-false}" = true ]; then
    Run cd ~/Library/Fonts
    # NOTE: https://github.com/edihbrandon/RictyDiminished/tree/55dd719ac9a57db3c0022866b783a092d0514037
    Run curl -LOR https://github.com/edihbrandon/RictyDiminished/raw/55dd719ac9a57db3c0022866b783a092d0514037/RictyDiminished-Bold.ttf
    Run curl -LOR https://github.com/edihbrandon/RictyDiminished/raw/55dd719ac9a57db3c0022866b783a092d0514037/RictyDiminished-BoldOblique.ttf
    Run curl -LOR https://github.com/edihbrandon/RictyDiminished/raw/55dd719ac9a57db3c0022866b783a092d0514037/RictyDiminished-Oblique.ttf
    Run curl -LOR https://github.com/edihbrandon/RictyDiminished/raw/55dd719ac9a57db3c0022866b783a092d0514037/RictyDiminished-Regular.ttf
  fi
)} && __main "$@"
